00099 /* 
00100  * Microsoft's C Runtime library is missing strcasecmp and strncasecmp. 
00101  * Other Win32 C runtime libraries, notably minwg, may define it. 
00102  * There is no symbol uniquely defined in Microsoft's header files that 
00103  * can be used by the preprocessor to know whether we're compiling for
00104  * Microsoft's library or not (or which version).  Thus there's no
00105  * way to automatically decide whether or not to define strcasecmp
00106  * in terms of stricmp.  
00107  * 
00108  * The Microsoft *compiler* defines _MSC_VER.  On the assumption that
00109  * anyone using their compiler is also using their library, the below 
00110  * tests check _MSC_VER as a proxy. 
00111  */
00112 #if defined(_WIN32)
00113 # if !defined(strcasecmp) && defined(_MSC_VER) 
00114 #     define  strcasecmp(A, B) stricmp((A), (B))
00115 # endif
00116 # if !defined(strncasecmp) && defined(_MSC_VER)
00117 #     define  strncasecmp(x,y,z) strnicmp((x),(y),(z))
00118 # endif
